    Create a new observable

    example
    
// Create an observable
const myObs = Observable.create(observer => {
  let x = 0;
  const y = setInterval(() => {
    // that emits incrementing values
    observer.next(x++);
  }, 1000); // each second
  return function cleanup() {
    // and cleans up after its self
    clearInterval(y);
  }
});
